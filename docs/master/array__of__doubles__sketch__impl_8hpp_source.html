70 template<typename A>
71 void compact_array_of_doubles_sketch_alloc<A>::serialize(std::ostream& os) const {
72  const uint8_t preamble_longs = 1;
73  write(os, preamble_longs);
74  const uint8_t serial_version = SERIAL_VERSION;
75  write(os, serial_version);
76  const uint8_t family = SKETCH_FAMILY;
77  write(os, family);
78  const uint8_t type = SKETCH_TYPE;
79  write(os, type);
80  const uint8_t flags_byte(
81  (this->is_empty() ? 1 << flags::IS_EMPTY : 0) |
82  (this->get_num_retained() > 0 ? 1 << flags::HAS_ENTRIES : 0) |
83  (this->is_ordered() ? 1 << flags::IS_ORDERED : 0)
84  );
85  write(os, flags_byte);
86  write(os, num_values_);
87  const uint16_t seed_hash = this->get_seed_hash();
88  write(os, seed_hash);
89  write(os, this->theta_);
90  if (this->get_num_retained() > 0) {
91  const uint32_t num_entries = static_cast<uint32_t>(this->entries_.size());
92  write(os, num_entries);
93  const uint32_t unused32 = 0;
94  write(os, unused32);
95  for (const auto& it: this->entries_) {
96  write(os, it.first);
97  }
98  for (const auto& it: this->entries_) {
99  write(os, it.second.data(), it.second.size() * sizeof(double));
100  }
101  }
102 }

104 template<typename A>
105 auto compact_array_of_doubles_sketch_alloc<A>::serialize(unsigned header_size_bytes) const -> vector_bytes {
106  const uint8_t preamble_longs = 1;
107  const size_t size = header_size_bytes + 16 // preamble and theta
108  + (this->entries_.size() > 0 ? 8 : 0)
109  + (sizeof(uint64_t) + sizeof(double) * num_values_) * this->entries_.size();
110  vector_bytes bytes(size, 0, this->entries_.get_allocator());
111  uint8_t* ptr = bytes.data() + header_size_bytes;
112 
113  ptr += copy_to_mem(preamble_longs, ptr);
114  const uint8_t serial_version = SERIAL_VERSION;
115  ptr += copy_to_mem(serial_version, ptr);
116  const uint8_t family = SKETCH_FAMILY;
117  ptr += copy_to_mem(family, ptr);
118  const uint8_t type = SKETCH_TYPE;
119  ptr += copy_to_mem(type, ptr);
120  const uint8_t flags_byte(
121  (this->is_empty() ? 1 << flags::IS_EMPTY : 0) |
122  (this->get_num_retained() ? 1 << flags::HAS_ENTRIES : 0) |
123  (this->is_ordered() ? 1 << flags::IS_ORDERED : 0)
124  );
125  ptr += copy_to_mem(flags_byte, ptr);
126  ptr += copy_to_mem(num_values_, ptr);
127  const uint16_t seed_hash = this->get_seed_hash();
128  ptr += copy_to_mem(seed_hash, ptr);
129  ptr += copy_to_mem((this->theta_), ptr);
130  if (this->get_num_retained() > 0) {
131  const uint32_t num_entries = static_cast<uint32_t>(this->entries_.size());
132  ptr += copy_to_mem(num_entries, ptr);
133  ptr += sizeof(uint32_t); // unused
134  for (const auto& it: this->entries_) {
135  ptr += copy_to_mem(it.first, ptr);
136  }
137  for (const auto& it: this->entries_) {
138  ptr += copy_to_mem(it.second.data(), ptr, it.second.size() * sizeof(double));
139  }
140  }
141  return bytes;
142 }

144 template<typename A>
145 compact_array_of_doubles_sketch_alloc<A> compact_array_of_doubles_sketch_alloc<A>::deserialize(std::istream& is, uint64_t seed, const A& allocator) {
146  read<uint8_t>(is); // unused
147  const auto serial_version = read<uint8_t>(is);
148  const auto family = read<uint8_t>(is);
149  const auto type = read<uint8_t>(is);
150  const auto flags_byte = read<uint8_t>(is);
151  const auto num_values = read<uint8_t>(is);
152  const auto seed_hash = read<uint16_t>(is);
153  checker<true>::check_serial_version(serial_version, SERIAL_VERSION);
154  checker<true>::check_sketch_family(family, SKETCH_FAMILY);
155  checker<true>::check_sketch_type(type, SKETCH_TYPE);
156  const bool has_entries = flags_byte & (1 << flags::HAS_ENTRIES);
157  if (has_entries) checker<true>::check_seed_hash(seed_hash, compute_seed_hash(seed));
158 
159  const auto theta = read<uint64_t>(is);
160  std::vector<Entry, AllocEntry> entries(allocator);
161  if (has_entries) {
162  const auto num_entries = read<uint32_t>(is);
163  read<uint32_t>(is); // unused
164  entries.reserve(num_entries);
165  std::vector<uint64_t, AllocU64> keys(num_entries, 0, allocator);
166  read(is, keys.data(), num_entries * sizeof(uint64_t));
167  for (size_t i = 0; i < num_entries; ++i) {
168  aod<A> summary(num_values, allocator);
169  read(is, summary.data(), num_values * sizeof(double));
170  entries.push_back(Entry(keys[i], std::move(summary)));
171  }
172  }
173  if (!is.good()) throw std::runtime_error("error reading from std::istream");
174  const bool is_empty = flags_byte & (1 << flags::IS_EMPTY);
175  const bool is_ordered = flags_byte & (1 << flags::IS_ORDERED);
176  return compact_array_of_doubles_sketch_alloc(is_empty, is_ordered, seed_hash, theta, std::move(entries), num_values);
177 }

179 template<typename A>
180 compact_array_of_doubles_sketch_alloc<A> compact_array_of_doubles_sketch_alloc<A>::deserialize(const void* bytes, size_t size, uint64_t seed, const A& allocator) {
181  ensure_minimum_memory(size, 16);
182  const char* ptr = static_cast<const char*>(bytes);
183  ptr += sizeof(uint8_t); // unused
184  uint8_t serial_version;
185  ptr += copy_from_mem(ptr, serial_version);
186  uint8_t family;
187  ptr += copy_from_mem(ptr, family);
188  uint8_t type;
189  ptr += copy_from_mem(ptr, type);
190  uint8_t flags_byte;
191  ptr += copy_from_mem(ptr, flags_byte);
192  uint8_t num_values;
193  ptr += copy_from_mem(ptr, num_values);
194  uint16_t seed_hash;
195  ptr += copy_from_mem(ptr, seed_hash);
196  checker<true>::check_serial_version(serial_version, SERIAL_VERSION);
197  checker<true>::check_sketch_family(family, SKETCH_FAMILY);
198  checker<true>::check_sketch_type(type, SKETCH_TYPE);
199  const bool has_entries = flags_byte & (1 << flags::HAS_ENTRIES);
200  if (has_entries) checker<true>::check_seed_hash(seed_hash, compute_seed_hash(seed));
201 
202  uint64_t theta;
203  ptr += copy_from_mem(ptr, theta);
204  std::vector<Entry, AllocEntry> entries(allocator);
205  if (has_entries) {
206  ensure_minimum_memory(size, 24);
207  uint32_t num_entries;
208  ptr += copy_from_mem(ptr, num_entries);
209  ptr += sizeof(uint32_t); // unused
210  ensure_minimum_memory(size, 24 + (sizeof(uint64_t) + sizeof(double) * num_values) * num_entries);
211  entries.reserve(num_entries);
212  std::vector<uint64_t, AllocU64> keys(num_entries, 0, allocator);
213  ptr += copy_from_mem(ptr, keys.data(), sizeof(uint64_t) * num_entries);
214  for (size_t i = 0; i < num_entries; ++i) {
215  aod<A> summary(num_values, allocator);
216  ptr += copy_from_mem(ptr, summary.data(), num_values * sizeof(double));
217  entries.push_back(Entry(keys[i], std::move(summary)));
218  }
219  }
220  const bool is_empty = flags_byte & (1 << flags::IS_EMPTY);
221  const bool is_ordered = flags_byte & (1 << flags::IS_ORDERED);
222  return compact_array_of_doubles_sketch_alloc(is_empty, is_ordered, seed_hash, theta, std::move(entries), num_values);
223 }
