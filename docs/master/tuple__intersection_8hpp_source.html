58  // reformulate the external policy that operates on Summary

60  struct internal_policy {
61  internal_policy(const Policy& policy): policy_(policy) {}
62  void operator()(Entry& internal_entry, const Entry& incoming_entry) const {
63  policy_(internal_entry.second, incoming_entry.second);
64  }
65  void operator()(Entry& internal_entry, Entry&& incoming_entry) const {
66  policy_(internal_entry.second, std::move(incoming_entry.second));
67  }
68  const Policy& get_policy() const { return policy_; }

70  };
