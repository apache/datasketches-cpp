95 template<typename T>
96 T byteswap(T value) {
97  char* ptr = static_cast<char*>(static_cast<void*>(&value));
98  const int len = sizeof(T);
99  for (size_t i = 0; i < len / 2; ++i) {
100  std::swap(ptr[i], ptr[len - i - 1]);
101  }
102  return value;
103 }
