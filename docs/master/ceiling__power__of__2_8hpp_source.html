27 // compute the next highest power of 2 of 32-bit n
28 // taken from https://graphics.stanford.edu/~seander/bithacks.html
29 static inline uint32_t ceiling_power_of_2(uint32_t n) {
30  --n;
31  n |= n >> 1;
32  n |= n >> 2;
33  n |= n >> 4;
34  n |= n >> 8;
35  n |= n >> 16;
36  return ++n;
37 }
