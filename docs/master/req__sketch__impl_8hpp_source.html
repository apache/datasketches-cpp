186 template<typename T, typename C, typename A>
187 template<typename FwdT>
188 void req_sketch<T, C, A>::update(FwdT&& item) {
189  if (!check_update_item(item)) { return; }
190  if (is_empty()) {
191  min_item_ = new (allocator_.allocate(1)) T(item);
192  max_item_ = new (allocator_.allocate(1)) T(item);
193  } else {
194  if (comparator_(item, *min_item_)) *min_item_ = item;
195  if (comparator_(*max_item_, item)) *max_item_ = item;
196  }
197  compactors_[0].append(std::forward<FwdT>(item));
198  ++num_retained_;
199  ++n_;
200  if (num_retained_ == max_nom_size_) compress();
201  reset_sorted_view();
202 }

204 template<typename T, typename C, typename A>
205 template<typename FwdSk>
206 void req_sketch<T, C, A>::merge(FwdSk&& other) {
207  if (is_HRA() != other.is_HRA()) throw std::invalid_argument("merging HRA and LRA is not valid");
208  if (other.is_empty()) return;
209  if (is_empty()) {
210  min_item_ = new (allocator_.allocate(1)) T(conditional_forward<FwdSk>(*other.min_item_));
211  max_item_ = new (allocator_.allocate(1)) T(conditional_forward<FwdSk>(*other.max_item_));
212  } else {
213  if (comparator_(*other.min_item_, *min_item_)) *min_item_ = conditional_forward<FwdSk>(*other.min_item_);
214  if (comparator_(*max_item_, *other.max_item_)) *max_item_ = conditional_forward<FwdSk>(*other.max_item_);
215  }
216  // grow until this has at least as many compactors as other
217  while (get_num_levels() < other.get_num_levels()) grow();
218  // merge the items in all height compactors
219  for (size_t i = 0; i < other.get_num_levels(); ++i) {
220  compactors_[i].merge(conditional_forward<FwdSk>(other.compactors_[i]));
221  }
222  n_ += other.n_;
223  update_max_nom_size();
224  update_num_retained();
225  if (num_retained_ >= max_nom_size_) compress();
226  reset_sorted_view();
227 }

286 template<typename T, typename C, typename A>
287 std::vector<T, A> req_sketch<T, C, A>::get_quantiles(const double* ranks, uint32_t size, bool inclusive) const {
288  if (is_empty()) throw std::runtime_error("operation is undefined for an empty sketch");
289  std::vector<T, A> quantiles(allocator_);
290  quantiles.reserve(size);
291 
292  // possible side-effect of sorting level zero
293  setup_sorted_view();
294 
295  for (uint32_t i = 0; i < size; ++i) {
296  const double rank = ranks[i];
297  if ((rank < 0.0) || (rank > 1.0)) {
298  throw std::invalid_argument("Normalized rank cannot be less than 0 or greater than 1");
299  }
300  quantiles.push_back(sorted_view_->get_quantile(rank, inclusive));
301  }
302  return quantiles;

305 template<typename T, typename C, typename A>
306 quantiles_sorted_view<T, C, A> req_sketch<T, C, A>::get_sorted_view() const {
307  if (!compactors_[0].is_sorted()) {
308  const_cast<Compactor&>(compactors_[0]).sort(); // allow this side effect
309  }
310  quantiles_sorted_view<T, C, A> view(get_num_retained(), comparator_, allocator_);
311 
312  for (auto& compactor: compactors_) {
313  view.add(compactor.begin(), compactor.end(), 1ULL << compactor.get_lg_weight());
314  }
315 
316  view.convert_to_cummulative();
317  return view;
318 }

335 template<typename T, typename C, typename A>
336 double req_sketch<T, C, A>::get_rank_lb(uint16_t k, uint8_t num_levels, double rank, uint8_t num_std_dev, uint64_t n, bool hra) {
337  if (is_exact_rank(k, num_levels, rank, n, hra)) return rank;
338  const double relative = relative_rse_factor() / k * (hra ? 1.0 - rank : rank);
339  const double fixed = FIXED_RSE_FACTOR / k;
340  const double lb_rel = rank - num_std_dev * relative;
341  const double lb_fix = rank - num_std_dev * fixed;
342  return std::max(lb_rel, lb_fix);
343 }

345 template<typename T, typename C, typename A>
346 double req_sketch<T, C, A>::get_rank_ub(uint16_t k, uint8_t num_levels, double rank, uint8_t num_std_dev, uint64_t n, bool hra) {
347  if (is_exact_rank(k, num_levels, rank, n, hra)) return rank;
348  const double relative = relative_rse_factor() / k * (hra ? 1.0 - rank : rank);
349  const double fixed = FIXED_RSE_FACTOR / k;
350  const double ub_rel = rank + num_std_dev * relative;
351  const double ub_fix = rank + num_std_dev * fixed;
352  return std::min(ub_rel, ub_fix);
353 }

355 template<typename T, typename C, typename A>
356 bool req_sketch<T, C, A>::is_exact_rank(uint16_t k, uint8_t num_levels, double rank, uint64_t n, bool hra) {
357  const unsigned base_cap = k * req_constants::INIT_NUM_SECTIONS;
358  if (num_levels == 1 || n <= base_cap) return true;
359  const double exact_rank_thresh = static_cast<double>(base_cap) / n;
360  return (hra && rank >= 1.0 - exact_rank_thresh) || (!hra && rank <= exact_rank_thresh);
361 }

368 // implementation for fixed-size arithmetic types (integral and floating point)
369 template<typename T, typename C, typename A>
370 template<typename TT, typename SerDe, typename std::enable_if<std::is_arithmetic<TT>::value, int>::type>
371 size_t req_sketch<T, C, A>::get_serialized_size_bytes(const SerDe& sd) const {
372  size_t size = PREAMBLE_SIZE_BYTES;
373  if (is_empty()) return size;
374  if (is_estimation_mode()) {
375  size += sizeof(n_) + sizeof(TT) * 2; // min and max
376  }
377  if (n_ == 1) {
378  size += sizeof(TT);
379  } else {
380  for (const auto& compactor: compactors_) size += compactor.get_serialized_size_bytes(sd);
381  }
382  return size;
383 }

385 // implementation for all other types
386 template<typename T, typename C, typename A>
387 template<typename TT, typename SerDe, typename std::enable_if<!std::is_arithmetic<TT>::value, int>::type>
388 size_t req_sketch<T, C, A>::get_serialized_size_bytes(const SerDe& sd) const {
389  size_t size = PREAMBLE_SIZE_BYTES;
390  if (is_empty()) return size;
391  if (is_estimation_mode()) {
392  size += sizeof(n_);
393  size += sd.size_of_item(*min_item_);
394  size += sd.size_of_item(*max_item_);
395  }
396  if (n_ == 1) {
397  size += sd.size_of_item(*compactors_[0].begin());
398  } else {
399  for (const auto& compactor: compactors_) size += compactor.get_serialized_size_bytes(sd);
400  }
401  return size;
402 }

404 template<typename T, typename C, typename A>
405 template<typename SerDe>
406 void req_sketch<T, C, A>::serialize(std::ostream& os, const SerDe& sd) const {
407  const uint8_t preamble_ints = is_estimation_mode() ? 4 : 2;
408  write(os, preamble_ints);
409  const uint8_t serial_version = SERIAL_VERSION;
410  write(os, serial_version);
411  const uint8_t family = FAMILY;
412  write(os, family);
413  const bool raw_items = n_ <= req_constants::MIN_K;
414  const uint8_t flags_byte(
415  (is_empty() ? 1 << flags::IS_EMPTY : 0)
416  | (hra_ ? 1 << flags::IS_HIGH_RANK : 0)
417  | (raw_items ? 1 << flags::RAW_ITEMS : 0)
418  | (compactors_[0].is_sorted() ? 1 << flags::IS_LEVEL_ZERO_SORTED : 0)
419  );
420  write(os, flags_byte);
421  write(os, k_);
422  const uint8_t num_levels = is_empty() ? 0 : get_num_levels();
423  write(os, num_levels);
424  const uint8_t num_raw_items = raw_items ? static_cast<uint8_t>(n_) : 0;
425  write(os, num_raw_items);
426  if (is_empty()) return;
427  if (is_estimation_mode()) {
428  write(os, n_);
429  sd.serialize(os, min_item_, 1);
430  sd.serialize(os, max_item_, 1);
431  }
432  if (raw_items) {
433  sd.serialize(os, compactors_[0].begin(), num_raw_items);
434  } else {
435  for (const auto& compactor: compactors_) compactor.serialize(os, sd);
436  }
437 }

439 template<typename T, typename C, typename A>
440 template<typename SerDe>
441 auto req_sketch<T, C, A>::serialize(unsigned header_size_bytes, const SerDe& sd) const -> vector_bytes {
442  const size_t size = header_size_bytes + get_serialized_size_bytes(sd);
443  vector_bytes bytes(size, 0, allocator_);
444  uint8_t* ptr = bytes.data() + header_size_bytes;
445  const uint8_t* end_ptr = ptr + size;
446 
447  const uint8_t preamble_ints = is_estimation_mode() ? 4 : 2;
448  ptr += copy_to_mem(preamble_ints, ptr);
449  const uint8_t serial_version = SERIAL_VERSION;
450  ptr += copy_to_mem(serial_version, ptr);
451  const uint8_t family = FAMILY;
452  ptr += copy_to_mem(family, ptr);
453  const bool raw_items = n_ <= req_constants::MIN_K;
454  const uint8_t flags_byte(
455  (is_empty() ? 1 << flags::IS_EMPTY : 0)
456  | (hra_ ? 1 << flags::IS_HIGH_RANK : 0)
457  | (raw_items ? 1 << flags::RAW_ITEMS : 0)
458  | (compactors_[0].is_sorted() ? 1 << flags::IS_LEVEL_ZERO_SORTED : 0)
459  );
460  ptr += copy_to_mem(flags_byte, ptr);
461  ptr += copy_to_mem(k_, ptr);
462  const uint8_t num_levels = is_empty() ? 0 : get_num_levels();
463  ptr += copy_to_mem(num_levels, ptr);
464  const uint8_t num_raw_items = raw_items ? static_cast<uint8_t>(n_) : 0;
465  ptr += copy_to_mem(num_raw_items, ptr);
466  if (!is_empty()) {
467  if (is_estimation_mode()) {
468  ptr += copy_to_mem(n_, ptr);
469  ptr += sd.serialize(ptr, end_ptr - ptr, min_item_, 1);
470  ptr += sd.serialize(ptr, end_ptr - ptr, max_item_, 1);
471  }
472  if (raw_items) {
473  ptr += sd.serialize(ptr, end_ptr - ptr, compactors_[0].begin(), num_raw_items);
474  } else {
475  for (const auto& compactor: compactors_) ptr += compactor.serialize(ptr, end_ptr - ptr, sd);
476  }
477  }
478  return bytes;
479 }

481 template<typename T, typename C, typename A>
482 template<typename SerDe>
483 req_sketch<T, C, A> req_sketch<T, C, A>::deserialize(std::istream& is, const SerDe& sd, const C& comparator, const A& allocator) {
484  const auto preamble_ints = read<uint8_t>(is);
485  const auto serial_version = read<uint8_t>(is);
486  const auto family_id = read<uint8_t>(is);
487  const auto flags_byte = read<uint8_t>(is);
488  const auto k = read<uint16_t>(is);
489  const auto num_levels = read<uint8_t>(is);
490  const auto num_raw_items = read<uint8_t>(is);
491 
492  check_preamble_ints(preamble_ints, num_levels);
493  check_serial_version(serial_version);
494  check_family_id(family_id);
495 
496  if (!is.good()) throw std::runtime_error("error reading from std::istream");
497  const bool is_empty = flags_byte & (1 << flags::IS_EMPTY);
498  const bool hra = flags_byte & (1 << flags::IS_HIGH_RANK);
499  if (is_empty) return req_sketch(k, hra, comparator, allocator);
500 
501  A alloc(allocator);
502  auto item_buffer_deleter = [&alloc](T* ptr) { alloc.deallocate(ptr, 1); };
503  std::unique_ptr<T, decltype(item_buffer_deleter)> min_item_buffer(alloc.allocate(1), item_buffer_deleter);
504  std::unique_ptr<T, decltype(item_buffer_deleter)> max_item_buffer(alloc.allocate(1), item_buffer_deleter);
505  std::unique_ptr<T, item_deleter> min_item(nullptr, item_deleter(allocator));
506  std::unique_ptr<T, item_deleter> max_item(nullptr, item_deleter(allocator));
507 
508  const bool raw_items = flags_byte & (1 << flags::RAW_ITEMS);
509  const bool is_level_0_sorted = flags_byte & (1 << flags::IS_LEVEL_ZERO_SORTED);
510  std::vector<Compactor, AllocCompactor> compactors(allocator);
511 
512  uint64_t n = 1;
513  if (num_levels > 1) {
514  n = read<uint64_t>(is);
515  sd.deserialize(is, min_item_buffer.get(), 1);
516  // serde call did not throw, repackage with destrtuctor
517  min_item = std::unique_ptr<T, item_deleter>(min_item_buffer.release(), item_deleter(allocator));
518  sd.deserialize(is, max_item_buffer.get(), 1);
519  // serde call did not throw, repackage with destrtuctor
520  max_item = std::unique_ptr<T, item_deleter>(max_item_buffer.release(), item_deleter(allocator));
521  }
522 
523  if (raw_items) {
524  compactors.push_back(Compactor::deserialize(is, sd, comparator, allocator, is_level_0_sorted, k, num_raw_items, hra));
525  } else {
526  for (size_t i = 0; i < num_levels; ++i) {
527  compactors.push_back(Compactor::deserialize(is, sd, comparator, allocator, i == 0 ? is_level_0_sorted : true, hra));
528  }
529  }
530  if (num_levels == 1) {
531  const auto begin = compactors[0].begin();
532  const auto end = compactors[0].end();
533  n = compactors[0].get_num_items();
534  auto min_it = begin;
535  auto max_it = begin;
536  for (auto it = begin; it != end; ++it) {
537  if (comparator(*it, *min_it)) min_it = it;
538  if (comparator(*max_it, *it)) max_it = it;
539  }
540  new (min_item_buffer.get()) T(*min_it);
541  // copy did not throw, repackage with destrtuctor
542  min_item = std::unique_ptr<T, item_deleter>(min_item_buffer.release(), item_deleter(allocator));
543  new (max_item_buffer.get()) T(*max_it);
544  // copy did not throw, repackage with destrtuctor
545  max_item = std::unique_ptr<T, item_deleter>(max_item_buffer.release(), item_deleter(allocator));
546  }
547 
548  if (!is.good()) throw std::runtime_error("error reading from std::istream");
549  return req_sketch(k, hra, n, std::move(min_item), std::move(max_item), std::move(compactors), comparator);
550 }

552 template<typename T, typename C, typename A>
553 template<typename SerDe>
554 req_sketch<T, C, A> req_sketch<T, C, A>::deserialize(const void* bytes, size_t size, const SerDe& sd, const C& comparator, const A& allocator) {
555  ensure_minimum_memory(size, 8);
556  const char* ptr = static_cast<const char*>(bytes);
557  const char* end_ptr = static_cast<const char*>(bytes) + size;
558 
559  uint8_t preamble_ints;
560  ptr += copy_from_mem(ptr, preamble_ints);
561  uint8_t serial_version;
562  ptr += copy_from_mem(ptr, serial_version);
563  uint8_t family_id;
564  ptr += copy_from_mem(ptr, family_id);
565  uint8_t flags_byte;
566  ptr += copy_from_mem(ptr, flags_byte);
567  uint16_t k;
568  ptr += copy_from_mem(ptr, k);
569  uint8_t num_levels;
570  ptr += copy_from_mem(ptr, num_levels);
571  uint8_t num_raw_items;
572  ptr += copy_from_mem(ptr, num_raw_items);
573 
574  check_preamble_ints(preamble_ints, num_levels);
575  check_serial_version(serial_version);
576  check_family_id(family_id);
577 
578  const bool is_empty = flags_byte & (1 << flags::IS_EMPTY);
579  const bool hra = flags_byte & (1 << flags::IS_HIGH_RANK);
580  if (is_empty) return req_sketch(k, hra, comparator, allocator);
581 
582  A alloc(allocator);
583  auto item_buffer_deleter = [&alloc](T* ptr) { alloc.deallocate(ptr, 1); };
584  std::unique_ptr<T, decltype(item_buffer_deleter)> min_item_buffer(alloc.allocate(1), item_buffer_deleter);
585  std::unique_ptr<T, decltype(item_buffer_deleter)> max_item_buffer(alloc.allocate(1), item_buffer_deleter);
586  std::unique_ptr<T, item_deleter> min_item(nullptr, item_deleter(allocator));
587  std::unique_ptr<T, item_deleter> max_item(nullptr, item_deleter(allocator));
588 
589  const bool raw_items = flags_byte & (1 << flags::RAW_ITEMS);
590  const bool is_level_0_sorted = flags_byte & (1 << flags::IS_LEVEL_ZERO_SORTED);
591  std::vector<Compactor, AllocCompactor> compactors(allocator);
592 
593  uint64_t n = 1;
594  if (num_levels > 1) {
595  ensure_minimum_memory(end_ptr - ptr, sizeof(n));
596  ptr += copy_from_mem(ptr, n);
597  ptr += sd.deserialize(ptr, end_ptr - ptr, min_item_buffer.get(), 1);
598  // serde call did not throw, repackage with destrtuctor
599  min_item = std::unique_ptr<T, item_deleter>(min_item_buffer.release(), item_deleter(allocator));
600  ptr += sd.deserialize(ptr, end_ptr - ptr, max_item_buffer.get(), 1);
601  // serde call did not throw, repackage with destrtuctor
602  max_item = std::unique_ptr<T, item_deleter>(max_item_buffer.release(), item_deleter(allocator));
603  }
604 
605  if (raw_items) {
606  auto pair = Compactor::deserialize(ptr, end_ptr - ptr, sd, comparator, allocator, is_level_0_sorted, k, num_raw_items, hra);
607  compactors.push_back(std::move(pair.first));
608  ptr += pair.second;
609  } else {
610  for (size_t i = 0; i < num_levels; ++i) {
611  auto pair = Compactor::deserialize(ptr, end_ptr - ptr, sd, comparator, allocator, i == 0 ? is_level_0_sorted : true, hra);
612  compactors.push_back(std::move(pair.first));
613  ptr += pair.second;
614  }
615  }
616  if (num_levels == 1) {
617  const auto begin = compactors[0].begin();
618  const auto end = compactors[0].end();
619  n = compactors[0].get_num_items();
620  auto min_it = begin;
621  auto max_it = begin;
622  for (auto it = begin; it != end; ++it) {
623  if (comparator(*it, *min_it)) min_it = it;
624  if (comparator(*max_it, *it)) max_it = it;
625  }
626  new (min_item_buffer.get()) T(*min_it);
627  // copy did not throw, repackage with destrtuctor
628  min_item = std::unique_ptr<T, item_deleter>(min_item_buffer.release(), item_deleter(allocator));
629  new (max_item_buffer.get()) T(*max_it);
630  // copy did not throw, repackage with destrtuctor
631  max_item = std::unique_ptr<T, item_deleter>(max_item_buffer.release(), item_deleter(allocator));
632  }
633 
634  return req_sketch(k, hra, n, std::move(min_item), std::move(max_item), std::move(compactors), comparator);
635 }

661 template<typename T, typename C, typename A>
662 void req_sketch<T, C, A>::compress() {
663  for (size_t h = 0; h < compactors_.size(); ++h) {
664  if (compactors_[h].get_num_items() >= compactors_[h].get_nom_capacity()) {
665  if (h == 0) compactors_[0].sort();
666  if (h + 1 >= get_num_levels()) { // at the top?
667  grow(); // add a level, increases max_nom_size
668  }
669  auto pair = compactors_[h].compact(compactors_[h + 1]);
670  num_retained_ -= pair.first;
671  max_nom_size_ += pair.second;
672  if (LAZY_COMPRESSION && num_retained_ < max_nom_size_) break;
673  }
674  }
675 }
