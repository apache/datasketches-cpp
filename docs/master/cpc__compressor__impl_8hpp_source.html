52 template<typename A>
53 uint8_t* cpc_compressor<A>::make_inverse_permutation(const uint8_t* permu, unsigned length) {
54  uint8_t* inverse = new uint8_t[length]; // use new for global initialization
55  for (unsigned i = 0; i < length; i++) {
56  inverse[permu[i]] = static_cast<uint8_t>(i);
57  }
58  for (unsigned i = 0; i < length; i++) {
59  if (permu[inverse[i]] != i) throw std::logic_error("inverse permutation error");
60  }
61  return inverse;
62 }

64 /* Given an encoding table that maps unsigned bytes to codewords
65  of length at most 12, this builds a size-4096 decoding table */
66 // The second argument is typically 256, but can be other values such as 65.
67 template<typename A>
68 uint16_t* cpc_compressor<A>::make_decoding_table(const uint16_t* encoding_table, unsigned num_byte_values) {
69  uint16_t* decoding_table = new uint16_t[4096]; // use new for global initialization
70  for (unsigned byte_value = 0; byte_value < num_byte_values; byte_value++) {
71  const uint16_t encoding_entry = encoding_table[byte_value];
72  const uint16_t code_value = encoding_entry & 0xfff;
73  const uint8_t code_length = encoding_entry >> 12;
74  const uint16_t decoding_entry = static_cast<uint16_t>((code_length << 8) | byte_value);
75  const uint8_t garbage_length = 12 - code_length;
76  const uint32_t num_copies = 1 << garbage_length;
77  for (uint32_t garbage_bits = 0; garbage_bits < num_copies; garbage_bits++) {
78  const uint16_t extended_code_value = static_cast<uint16_t>(code_value | (garbage_bits << code_length));
79  decoding_table[extended_code_value & 0xfff] = decoding_entry;
80  }
81  }
82  return decoding_table;
83 }

85 template<typename A>
86 void cpc_compressor<A>::validate_decoding_table(const uint16_t* decoding_table, const uint16_t* encoding_table) const {
87  for (int decode_this = 0; decode_this < 4096; decode_this++) {
88  const int tmp_d = decoding_table[decode_this];
89  const int decoded_byte = tmp_d & 0xff;
90  const int decoded_length = tmp_d >> 8;
91 
92  const int tmp_e = encoding_table[decoded_byte];
93  const int encoded_bit_pattern = tmp_e & 0xfff;
94  const int encoded_length = tmp_e >> 12;
95 
96  if (decoded_length != encoded_length) throw std::logic_error("decoded length error");
97  if (encoded_bit_pattern != (decode_this & ((1 << decoded_length) - 1))) throw std::logic_error("bit pattern error");
98  }
99 }

102 void cpc_compressor<A>::make_decoding_tables() {
103  length_limited_unary_decoding_table65 = make_decoding_table(length_limited_unary_encoding_table65, 65);
104  validate_decoding_table(
105  length_limited_unary_decoding_table65,
106  length_limited_unary_encoding_table65
107  );
108 
109  for (int i = 0; i < (16 + 6); i++) {
110  decoding_tables_for_high_entropy_byte[i] = make_decoding_table(encoding_tables_for_high_entropy_byte[i], 256);

113  encoding_tables_for_high_entropy_byte[i]
114  );
115  }
116 
117  for (int i = 0; i < 16; i++) {
118  column_permutations_for_decoding[i] = make_inverse_permutation(column_permutations_for_encoding[i], 56);

133 template<typename A>
134 void cpc_compressor<A>::compress(const cpc_sketch_alloc<A>& source, compressed_state<A>& result) const {
135  switch (source.determine_flavor()) {
136  case cpc_sketch_alloc<A>::flavor::EMPTY:
137  break;
138  case cpc_sketch_alloc<A>::flavor::SPARSE:
139  compress_sparse_flavor(source, result);
140  if (result.window_data.size() > 0) throw std::logic_error("window is not expected");
141  if (result.table_data.size() == 0) throw std::logic_error("table is expected");
142  break;
143  case cpc_sketch_alloc<A>::flavor::HYBRID:
144  compress_hybrid_flavor(source, result);
145  if (result.window_data.size() > 0) throw std::logic_error("window is not expected");
146  if (result.table_data.size() == 0) throw std::logic_error("table is expected");
147  break;
148  case cpc_sketch_alloc<A>::flavor::PINNED:
149  compress_pinned_flavor(source, result);
150  if (result.window_data.size() == 0) throw std::logic_error("window is not expected");
151  break;
152  case cpc_sketch_alloc<A>::flavor::SLIDING:
153  compress_sliding_flavor(source, result);
154  if (result.window_data.size() == 0) throw std::logic_error("window is expected");
155  break;
156  default: throw std::logic_error("Unknown sketch flavor");
157  }
158 }

160 template<typename A>
161 void cpc_compressor<A>::uncompress(const compressed_state<A>& source, uncompressed_state<A>& target, uint8_t lg_k, uint32_t num_coupons) const {
162  switch (cpc_sketch_alloc<A>::determine_flavor(lg_k, num_coupons)) {
163  case cpc_sketch_alloc<A>::flavor::EMPTY:
164  target.table = u32_table<A>(2, 6 + lg_k, source.table_data.get_allocator());
165  break;
166  case cpc_sketch_alloc<A>::flavor::SPARSE:
167  uncompress_sparse_flavor(source, target, lg_k);
168  break;
169  case cpc_sketch_alloc<A>::flavor::HYBRID:
170  uncompress_hybrid_flavor(source, target, lg_k);
171  break;
172  case cpc_sketch_alloc<A>::flavor::PINNED:
173  if (source.window_data.size() == 0) throw std::logic_error("window is expected");
174  uncompress_pinned_flavor(source, target, lg_k, num_coupons);
175  break;
176  case cpc_sketch_alloc<A>::flavor::SLIDING:
177  uncompress_sliding_flavor(source, target, lg_k, num_coupons);
178  break;
179  default: std::logic_error("Unknown sketch flavor");
180  }
181 }

184 void cpc_compressor<A>::compress_sparse_flavor(const cpc_sketch_alloc<A>& source, compressed_state<A>& result) const {
185  if (source.sliding_window.size() > 0) throw std::logic_error("unexpected sliding window");
186  vector_u32<A> pairs = source.surprising_value_table.unwrapping_get_items();
187  u32_table<A>::introspective_insertion_sort(pairs.data(), 0, pairs.size());

189 }

191 template<typename A>
192 void cpc_compressor<A>::uncompress_sparse_flavor(const compressed_state<A>& source, uncompressed_state<A>& target, uint8_t lg_k) const {
193  if (source.window_data.size() > 0) throw std::logic_error("unexpected sliding window");
194  if (source.table_data.size() == 0) throw std::logic_error("table is expected");
195  vector_u32<A> pairs = uncompress_surprising_values(source.table_data.data(), source.table_data_words, source.table_num_entries,
196  lg_k, source.table_data.get_allocator());

200 // This is complicated because it effectively builds a Sparse version
201 // of a Pinned sketch before compressing it. Hence the name Hybrid.
202 template<typename A>
203 void cpc_compressor<A>::compress_hybrid_flavor(const cpc_sketch_alloc<A>& source, compressed_state<A>& result) const {
204  if (source.sliding_window.size() == 0) throw std::logic_error("no sliding window");
205  if (source.window_offset != 0) throw std::logic_error("window_offset != 0");
206  const uint32_t k = 1 << source.get_lg_k();
207  vector_u32<A> pairs_from_table = source.surprising_value_table.unwrapping_get_items();
208  const uint32_t num_pairs_from_table = static_cast<uint32_t>(pairs_from_table.size());
209  if (num_pairs_from_table > 0) u32_table<A>::introspective_insertion_sort(pairs_from_table.data(), 0, num_pairs_from_table);
210  const uint32_t num_pairs_from_window = source.get_num_coupons() - num_pairs_from_table; // because the window offset is zero
211 
212  vector_u32<A> all_pairs = tricky_get_pairs_from_window(source.sliding_window.data(), k, num_pairs_from_window, num_pairs_from_table, source.get_allocator());
213 
214  u32_table<A>::merge(
215  pairs_from_table.data(), 0, pairs_from_table.size(),
216  all_pairs.data(), num_pairs_from_table, num_pairs_from_window,
217  all_pairs.data(), 0
218  ); // note the overlapping subarray trick
219 
220  compress_surprising_values(all_pairs, source.get_lg_k(), result);
221 }

223 template<typename A>
224 void cpc_compressor<A>::uncompress_hybrid_flavor(const compressed_state<A>& source, uncompressed_state<A>& target, uint8_t lg_k) const {
225  if (source.window_data.size() > 0) throw std::logic_error("window is not expected");
226  if (source.table_data.size() == 0) throw std::logic_error("table is expected");
227  vector_u32<A> pairs = uncompress_surprising_values(source.table_data.data(), source.table_data_words, source.table_num_entries,
228  lg_k, source.table_data.get_allocator());
229 
230  // In the hybrid flavor, some of these pairs actually
231  // belong in the window, so we will separate them out,
232  // moving the "true" pairs to the bottom of the array.
233  const uint32_t k = 1 << lg_k;
234  target.window.resize(k, 0); // important: zero the memory
235  uint32_t next_true_pair = 0;
236  for (uint32_t i = 0; i < source.table_num_entries; i++) {
237  const uint32_t row_col = pairs[i];
238  if (row_col == UINT32_MAX) throw std::logic_error("empty marker is not expected");
239  const uint8_t col = row_col & 63;
240  if (col < 8) {
241  const uint32_t row = row_col >> 6;
242  target.window[row] |= 1 << col; // set the window bit
243  } else {
244  pairs[next_true_pair++] = row_col; // move true pair down
245  }
246  }
247  target.table = u32_table<A>::make_from_pairs(pairs.data(), next_true_pair, lg_k, pairs.get_allocator());
248 }

250 template<typename A>
251 void cpc_compressor<A>::compress_pinned_flavor(const cpc_sketch_alloc<A>& source, compressed_state<A>& result) const {
252  compress_sliding_window(source.sliding_window.data(), source.get_lg_k(), source.get_num_coupons(), result);
253  vector_u32<A> pairs = source.surprising_value_table.unwrapping_get_items();
254  if (pairs.size() > 0) {
255  // Here we subtract 8 from the column indices. Because they are stored in the low 6 bits
256  // of each row_col pair, and because no column index is less than 8 for a "Pinned" sketch,
257  // we can simply subtract 8 from the pairs themselves.
258 
259  // shift the columns over by 8 positions before compressing (because of the window)
260  for (size_t i = 0; i < pairs.size(); i++) {
261  if ((pairs[i] & 63) < 8) throw std::logic_error("(pairs[i] & 63) < 8");
262  pairs[i] -= 8;
263  }
264 
265  if (pairs.size() > 0) u32_table<A>::introspective_insertion_sort(pairs.data(), 0, pairs.size());
266  compress_surprising_values(pairs, source.get_lg_k(), result);
267  }
268 }

270 template<typename A>
271 void cpc_compressor<A>::uncompress_pinned_flavor(const compressed_state<A>& source, uncompressed_state<A>& target,
272  uint8_t lg_k, uint32_t num_coupons) const {
273  if (source.window_data.size() == 0) throw std::logic_error("window is expected");
274  uncompress_sliding_window(source.window_data.data(), source.window_data_words, target.window, lg_k, num_coupons);
275  const uint32_t num_pairs = source.table_num_entries;
276  if (num_pairs == 0) {
277  target.table = u32_table<A>(2, 6 + lg_k, source.table_data.get_allocator());
278  } else {
279  if (source.table_data.size() == 0) throw std::logic_error("table is expected");
280  vector_u32<A> pairs = uncompress_surprising_values(source.table_data.data(), source.table_data_words, num_pairs,
281  lg_k, source.table_data.get_allocator());
282  // undo the compressor's 8-column shift
283  for (uint32_t i = 0; i < num_pairs; i++) {
284  if ((pairs[i] & 63) >= 56) throw std::logic_error("(pairs[i] & 63) >= 56");
285  pairs[i] += 8;
286  }
287  target.table = u32_table<A>::make_from_pairs(pairs.data(), num_pairs, lg_k, pairs.get_allocator());
288  }
289 }

291 template<typename A>
292 void cpc_compressor<A>::compress_sliding_flavor(const cpc_sketch_alloc<A>& source, compressed_state<A>& result) const {
293  compress_sliding_window(source.sliding_window.data(), source.get_lg_k(), source.get_num_coupons(), result);
294  vector_u32<A> pairs = source.surprising_value_table.unwrapping_get_items();
295  if (pairs.size() > 0) {
296  // Here we apply a complicated transformation to the column indices, which
297  // changes the implied ordering of the pairs, so we must do it before sorting.
298 
299  const uint8_t pseudo_phase = determine_pseudo_phase(source.get_lg_k(), source.get_num_coupons());
300  if (pseudo_phase >= 16) throw std::logic_error("unexpected pseudo phase for sliding flavor");
301  const uint8_t* permutation = column_permutations_for_encoding[pseudo_phase];
302 
303  const uint8_t offset = source.window_offset;
304  if (offset > 56) throw std::out_of_range("offset out of range");
305 
306  for (size_t i = 0; i < pairs.size(); i++) {
307  const uint32_t row_col = pairs[i];
308  const uint32_t row = row_col >> 6;
309  uint8_t col = row_col & 63;
310  // first rotate the columns into a canonical configuration: new = ((old - (offset+8)) + 64) mod 64
311  col = (col + 56 - offset) & 63;
312  if (col >= 56) throw std::out_of_range("col out of range");
313  // then apply the permutation
314  col = permutation[col];
315  pairs[i] = (row << 6) | col;
316  }
317 
318  if (pairs.size() > 0) u32_table<A>::introspective_insertion_sort(pairs.data(), 0, pairs.size());
319  compress_surprising_values(pairs, source.get_lg_k(), result);
320  }
321 }

323 template<typename A>
324 void cpc_compressor<A>::uncompress_sliding_flavor(const compressed_state<A>& source, uncompressed_state<A>& target,
325  uint8_t lg_k, uint32_t num_coupons) const {
326  if (source.window_data.size() == 0) throw std::logic_error("window is expected");
327  uncompress_sliding_window(source.window_data.data(), source.window_data_words, target.window, lg_k, num_coupons);
328  const uint32_t num_pairs = source.table_num_entries;
329  if (num_pairs == 0) {
330  target.table = u32_table<A>(2, 6 + lg_k, source.table_data.get_allocator());
331  } else {
332  if (source.table_data.size() == 0) throw std::logic_error("table is expected");
333  vector_u32<A> pairs = uncompress_surprising_values(source.table_data.data(), source.table_data_words, num_pairs,
334  lg_k, source.table_data.get_allocator());
335 
336  const uint8_t pseudo_phase = determine_pseudo_phase(lg_k, num_coupons);
337  if (pseudo_phase >= 16) throw std::logic_error("unexpected pseudo phase for sliding flavor");
338  const uint8_t* permutation = column_permutations_for_decoding[pseudo_phase];
339 
340  uint8_t offset = cpc_sketch_alloc<A>::determine_correct_offset(lg_k, num_coupons);
341  if (offset > 56) throw std::out_of_range("offset out of range");
342 
343  for (uint32_t i = 0; i < num_pairs; i++) {
344  const uint32_t row_col = pairs[i];
345  const uint32_t row = row_col >> 6;
346  uint8_t col = row_col & 63;
347  // first undo the permutation
348  col = permutation[col];
349  // then undo the rotation: old = (new + (offset+8)) mod 64
350  col = (col + (offset + 8)) & 63;
351  pairs[i] = (row << 6) | col;
352  }
353 
354  target.table = u32_table<A>::make_from_pairs(pairs.data(), num_pairs, lg_k, pairs.get_allocator());
355  }
356 }

358 template<typename A>
359 void cpc_compressor<A>::compress_surprising_values(const vector_u32<A>& pairs, uint8_t lg_k, compressed_state<A>& result) const {
360  const uint32_t k = 1 << lg_k;
361  const uint32_t num_pairs = static_cast<uint32_t>(pairs.size());
362  const uint8_t num_base_bits = golomb_choose_number_of_base_bits(k + num_pairs, num_pairs);
363  const uint64_t table_len = safe_length_for_compressed_pair_buf(k, num_pairs, num_base_bits);
364  result.table_data.resize(table_len);
365 
366  uint32_t csv_length = low_level_compress_pairs(pairs.data(), static_cast<uint32_t>(pairs.size()), num_base_bits, result.table_data.data());
367 
368  // At this point we could free the unused portion of the compression output buffer,
369  // but it is not necessary if it is temporary
370  // Note: realloc caused strange timing spikes for lgK = 11 and 12.
371 
372  result.table_data_words = csv_length;
373  result.table_num_entries = num_pairs;
374 }

377 vector_u32<A> cpc_compressor<A>::uncompress_surprising_values(const uint32_t* data, uint32_t data_words, uint32_t num_pairs,
378  uint8_t lg_k, const A& allocator) const {
379  const uint32_t k = 1 << lg_k;

386 template<typename A>
387 void cpc_compressor<A>::compress_sliding_window(const uint8_t* window, uint8_t lg_k, uint32_t num_coupons, compressed_state<A>& target) const {
388  const uint32_t k = 1 << lg_k;
389  const size_t window_buf_len = safe_length_for_compressed_window_buf(k);
390  target.window_data.resize(window_buf_len);
391  const uint8_t pseudo_phase = determine_pseudo_phase(lg_k, num_coupons);
392  size_t data_words = low_level_compress_bytes(window, k, encoding_tables_for_high_entropy_byte[pseudo_phase], target.window_data.data());
393 
394  // At this point we could free the unused portion of the compression output buffer,
395  // but it is not necessary if it is temporary
396  // Note: realloc caused strange timing spikes for lgK = 11 and 12.
397 
398  target.window_data_words = static_cast<uint32_t>(data_words);
399 }

410 template<typename A>
411 size_t cpc_compressor<A>::safe_length_for_compressed_pair_buf(uint32_t k, uint32_t num_pairs, uint8_t num_base_bits) {
412  // Long ybits = k + numPairs; // simpler and safer UB
413  // The following tighter UB on ybits is based on page 198
414  // of the textbook "Managing Gigabytes" by Witten, Moffat, and Bell.
415  // Notice that if numBaseBits == 0 it coincides with (k + numPairs).
416  const size_t ybits = num_pairs * (1 + num_base_bits) + (k >> num_base_bits);
417  const size_t xbits = 12 * num_pairs;
418  const size_t padding = num_base_bits > 10 ? 0 : 10 - num_base_bits;
419  return divide_longs_rounding_up(xbits + ybits + padding, 32);
420 }
421 
422 // Explanation of padding: we write
423 // 1) xdelta (huffman, provides at least 1 bit, requires 12-bit lookahead)
424 // 2) ydeltaGolombHi (unary, provides at least 1 bit, requires 8-bit lookahead)
425 // 3) ydeltaGolombLo (straight B bits).
426 // So the 12-bit lookahead is the tight constraint, but there are at least (2 + B) bits emitted,
427 // so we would be safe with max (0, 10 - B) bits of padding at the end of the bitstream.
428 template<typename A>
429 size_t cpc_compressor<A>::safe_length_for_compressed_window_buf(uint32_t k) { // measured in 32-bit words
430  const size_t bits = 12 * k + 11; // 11 bits of padding, due to 12-bit lookahead, with 1 bit certainly present.
431  return divide_longs_rounding_up(bits, 32);
432 }

434 template<typename A>
435 uint8_t cpc_compressor<A>::determine_pseudo_phase(uint8_t lg_k, uint32_t c) {
436  const uint32_t k = 1 << lg_k;
437  // This mid-range logic produces pseudo-phases. They are used to select encoding tables.
438  // The thresholds were chosen by hand after looking at plots of measured compression.
439  if (1000 * c < 2375 * k) {
440  if ( 4 * c < 3 * k) return 16 + 0; // mid-range table
441  else if ( 10 * c < 11 * k) return 16 + 1; // mid-range table
442  else if ( 100 * c < 132 * k) return 16 + 2; // mid-range table
443  else if ( 3 * c < 5 * k) return 16 + 3; // mid-range table
444  else if (1000 * c < 1965 * k) return 16 + 4; // mid-range table
445  else if (1000 * c < 2275 * k) return 16 + 5; // mid-range table
446  else return 6; // steady-state table employed before its actual phase
447  } else { // This steady-state logic produces true phases. They are used to select
448  // encoding tables, and also column permutations for the "Sliding" flavor.
449  if (lg_k < 4) throw std::logic_error("lgK < 4");
450  const size_t tmp = c >> (lg_k - 4);
451  const uint8_t phase = tmp & 15;
452  if (phase >= 16) throw std::out_of_range("wrong phase");
453  return phase;
454  }

473 // It is the caller's responsibility to ensure that the compressed_words array is long enough.
474 template<typename A>
475 uint32_t cpc_compressor<A>::low_level_compress_bytes(
476  const uint8_t* byte_array, // input
477  uint32_t num_bytes_to_encode,
478  const uint16_t* encoding_table,
479  uint32_t* compressed_words // output
480 ) const {
481  uint64_t bitbuf = 0; // bits are packed into this first, then are flushed to compressed_words
482  uint8_t bufbits = 0; // number of bits currently in bitbuf; must be between 0 and 31
483  uint32_t next_word_index = 0;
484 
485  for (uint32_t byte_index = 0; byte_index < num_bytes_to_encode; byte_index++) {
486  const uint16_t code_info = encoding_table[byte_array[byte_index]];
487  const uint64_t code_val = code_info & 0xfff;
488  const uint8_t code_len = code_info >> 12;

499  if (bufbits >= 32) throw std::logic_error("bufbits >= 32");
500  compressed_words[next_word_index++] = bitbuf & 0xffffffff;
501  bitbuf = 0; bufbits = 0; // not really necessary
502  }
503  return next_word_index;

557 template<typename A>
558 uint32_t cpc_compressor<A>::low_level_compress_pairs(
559  const uint32_t* pair_array, // input
560  uint32_t num_pairs_to_encode,
561  uint8_t num_base_bits,
562  uint32_t* compressed_words // output
563 ) const {
564  uint64_t bitbuf = 0;
565  uint8_t bufbits = 0;
566  uint32_t next_word_index = 0;
567  const uint64_t golomb_lo_mask = (1 << num_base_bits) - 1;
568  uint32_t predicted_row_index = 0;
569  uint8_t predicted_col_index = 0;
570 
571  for (uint32_t pair_index = 0; pair_index < num_pairs_to_encode; pair_index++) {
572  const uint32_t row_col = pair_array[pair_index];
573  const uint32_t row_index = row_col >> 6;
574  const uint8_t col_index = row_col & 63;
575 
576  if (row_index != predicted_row_index) predicted_col_index = 0;
577 
578  if (row_index < predicted_row_index) throw std::logic_error("row_index < predicted_row_index");
579  if (col_index < predicted_col_index) throw std::logic_error("col_index < predicted_col_index");
580 
581  const uint32_t y_delta = row_index - predicted_row_index;
582  const uint8_t x_delta = col_index - predicted_col_index;
583 
584  predicted_row_index = row_index;
585  predicted_col_index = col_index + 1;
586 
587  const uint16_t code_info = length_limited_unary_encoding_table65[x_delta];
588  const uint64_t code_val = code_info & 0xfff;
589  const uint8_t code_len = static_cast<uint8_t>(code_info >> 12);
590  bitbuf |= code_val << bufbits;
591  bufbits += code_len;
592  maybe_flush_bitbuf(bitbuf, bufbits, compressed_words, next_word_index);
593 
594  const uint64_t golomb_lo = y_delta & golomb_lo_mask;
595  const uint64_t golomb_hi = y_delta >> num_base_bits;
596 
597  write_unary(compressed_words, next_word_index, bitbuf, bufbits, golomb_hi);
598 
599  bitbuf |= golomb_lo << bufbits;
600  bufbits += num_base_bits;
601  maybe_flush_bitbuf(bitbuf, bufbits, compressed_words, next_word_index);
602  }
603 
604  // Pad the bitstream so that the decompressor's 12-bit peek can't overrun its input.
605  const uint8_t padding = (num_base_bits > 10) ? 0 : 10 - num_base_bits;
606  bufbits += padding;
607  maybe_flush_bitbuf(bitbuf, bufbits, compressed_words, next_word_index);
608 
609  if (bufbits > 0) { // We are done encoding now, so we flush the bit buffer
610  if (bufbits >= 32) throw std::logic_error("bufbits >= 32");
611  compressed_words[next_word_index++] = bitbuf & 0xffffffff;
612  bitbuf = 0; bufbits = 0; // not really necessary
613  }
614 
615  return next_word_index;
616 }

618 template<typename A>
619 void cpc_compressor<A>::low_level_uncompress_pairs(
620  uint32_t* pair_array, // output
621  uint32_t num_pairs_to_decode,
622  uint8_t num_base_bits,
623  const uint32_t* compressed_words, // input
624  uint32_t num_compressed_words
625 ) const {
626  uint32_t word_index = 0;
627  uint64_t bitbuf = 0;
628  uint8_t bufbits = 0;
629  const uint64_t golomb_lo_mask = (1 << num_base_bits) - 1;
630  uint32_t predicted_row_index = 0;
631  uint8_t predicted_col_index = 0;
632 
633  // for each pair we need to read:
634  // x_delta (12-bit length-limited unary)
635  // y_delta_hi (unary)
636  // y_delta_lo (basebits)
637 
638  for (uint32_t pair_index = 0; pair_index < num_pairs_to_decode; pair_index++) {
639  maybe_fill_bitbuf(bitbuf, bufbits, compressed_words, word_index, 12); // ensure 12 bits in bit buffer
640  const size_t peek12 = bitbuf & 0xfff;
641  const uint16_t lookup = length_limited_unary_decoding_table65[peek12];
642  const uint8_t code_word_length = lookup >> 8;
643  const int8_t x_delta = lookup & 0xff;
644  bitbuf >>= code_word_length;
645  bufbits -= code_word_length;
646 
647  const uint64_t golomb_hi = read_unary(compressed_words, word_index, bitbuf, bufbits);
648 
649  maybe_fill_bitbuf(bitbuf, bufbits, compressed_words, word_index, num_base_bits); // ensure num_base_bits in bit buffer
650  const uint64_t golomb_lo = bitbuf & golomb_lo_mask;
651  bitbuf >>= num_base_bits;
652  bufbits -= num_base_bits;
653  const int64_t y_delta = (golomb_hi << num_base_bits) | golomb_lo;
654 
655  // Now that we have x_delta and y_delta, we can compute the pair's row and column
656  if (y_delta > 0) predicted_col_index = 0;
657  const uint32_t row_index = static_cast<uint32_t>(predicted_row_index + y_delta);
658  const uint8_t col_index = predicted_col_index + x_delta;
659  const uint32_t row_col = (row_index << 6) | col_index;
660  pair_array[pair_index] = row_col;
661  predicted_row_index = row_index;
662  predicted_col_index = col_index + 1;
663  }
664  if (word_index > num_compressed_words) throw std::logic_error("word_index > num_compressed_words"); // check for buffer over-run
665 }

667 uint64_t read_unary(
668  const uint32_t* compressed_words,
669  uint32_t& next_word_index,
670  uint64_t& bitbuf,
671  uint8_t& bufbits
672 ) {
673  if (compressed_words == nullptr) throw std::logic_error("compressed_words == NULL");
674  size_t subtotal = 0;
675  while (true) {
676  maybe_fill_bitbuf(bitbuf, bufbits, compressed_words, next_word_index, 8); // ensure 8 bits in bit buffer
677 
678  const uint8_t peek8 = bitbuf & 0xff; // These 8 bits include either all or part of the Unary codeword
679  const uint8_t trailing_zeros = byte_trailing_zeros_table[peek8];
680 
681  if (trailing_zeros > 8) throw std::out_of_range("trailing_zeros out of range");
682  if (trailing_zeros < 8) {
683  bufbits -= 1 + trailing_zeros;
684  bitbuf >>= 1 + trailing_zeros;
685  return subtotal + trailing_zeros;
686  }
687  // The codeword was partial, so read some more
688  subtotal += 8;
689  bufbits -= 8;
690  bitbuf >>= 8;
691  }
692 }

694 void write_unary(
695  uint32_t* compressed_words,
696  uint32_t& next_word_index,
697  uint64_t& bitbuf,
698  uint8_t& bufbits,
699  uint64_t value
700 ) {
701  if (compressed_words == nullptr) throw std::logic_error("compressed_words == NULL");
702  if (bufbits > 31) throw std::out_of_range("bufbits out of range");
703 
704  uint64_t remaining = value;
705 
706  while (remaining >= 16) {
707  remaining -= 16;
708  // Here we output 16 zeros, but we don't need to physically write them into bitbuf
709  // because it already contains zeros in that region.
710  bufbits += 16; // Record the fact that 16 bits of output have occurred.
711  maybe_flush_bitbuf(bitbuf, bufbits, compressed_words, next_word_index);
712  }
713 
714  if (remaining > 15) throw std::out_of_range("remaining out of range");
715 
716  const uint64_t the_unary_code = 1ULL << remaining;
717  bitbuf |= the_unary_code << bufbits;
718  bufbits += static_cast<uint8_t>(remaining + 1);
719  maybe_flush_bitbuf(bitbuf, bufbits, compressed_words, next_word_index);
720 }
